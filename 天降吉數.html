  <script>

    let historyData = [];

    function parseNumbers(input) {

      return input.split(',').map(n => parseInt(n.trim())).filter(n => !isNaN(n));

    }

    function generateNumbers(source = '') {

      let include = parseNumbers(document.getElementById('includeNumbers').value);

      let exclude = parseNumbers(document.getElementById('excludeNumbers').value);

      let result = [...include];

      while (result.length < 6) {

        let num = Math.floor(Math.random() * 49) + 1;

        if (!result.includes(num) && !exclude.includes(num)) {

          result.push(num);

        }

      }

      result.sort((a, b) => a - b);

      showAlert(result);

      historyData.push(result);

    }

    function generateFromWord() {

      let word = document.getElementById('wordInput').value.trim();

      if (word === '') return;

      let base = word.charCodeAt(0);

      let result = [];

      while (result.length < 6) {

        let num = (base + Math.floor(Math.random() * 100)) % 49 + 1;

        if (!result.includes(num)) result.push(num);

      }

      result.sort((a, b) => a - b);

      showAlert(result);

      historyData.push(result);

    }

    function generateFromBirthday() {

      let birthday = document.getElementById('birthdayInput').value.trim();

      if (birthday.length < 6) return;

      let seed = parseInt(birthday.slice(-6));

      let result = [];

      while (result.length < 6) {

        let num = (seed + Math.floor(Math.random() * 100)) % 49 + 1;

        if (!result.includes(num)) result.push(num);

      }

      result.sort((a, b) => a - b);

      showAlert(result);

      historyData.push(result);

    }

    function showAlert(numbers) {

      const blessings = [

        "頭獎得主就是你！",

        "財神爺保佑你！",

        "好運連連，大獎在望！",

        "頭獎號碼就是這一組！",

        "祝您發發發！",

        "財運亨通、金銀財寶滾滾來！",

        "人要發財了，什麼都擋不住！",

        "中獎在即，請笑納！",

        "喜從天降，富貴臨門！"

      ];

      const box = document.getElementById('alertBox');

      box.innerHTML = `

        <p style="font-size: 36px; font-weight: bold;">選號結果：${numbers.join("、")}</p>

        <p style="font-size: 30px;">${blessings[Math.floor(Math.random() * blessings.length)]}</p>

        <button onclick="closeAlert()" style="font-size: 28px;">繼續選號</button>

      `;

      box.style.display = 'block';

    }

    function closeAlert() {

      document.getElementById('alertBox').style.display = 'none';

    }

    function showHistory() {

      const div = document.getElementById('history');

      div.innerHTML = '';

      historyData.forEach((group, i) => {

        div.innerHTML += `<input type='checkbox' id='h${i}'><label for='h${i}'> ${group.join("、")} </label><br>`;

      });

    }

    function deleteSelected() {

      const newHistory = [];

      historyData.forEach((group, i) => {

        if (!document.getElementById('h' + i).checked) {

          newHistory.push(group);

        }

      });

      historyData = newHistory;

      showHistory();

    }
